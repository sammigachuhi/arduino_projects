LED Blink Code

arduino

Copy and paste this code into your Arduino IDE or Web Editor

1#define LED 13 // the pin the LED is connected to
2
3void setup()   {
4  pinMode(LED, OUTPUT); // Declare the LED as an output
5}
6
7void loop()   {
8  digitalWrite(LED, HIGH); // Turn the LED on
9  delay(1000); // Wait for   1000 milliseconds (1 second)
10  digitalWrite(LED, LOW); // Turn the LED off
11   delay(1000); // Wait for 1000 milliseconds (1 second)
12}
